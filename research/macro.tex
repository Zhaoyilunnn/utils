  bottom=1mm,                % Padding below content

  right=1mm,                 % Right padding
  titlerule=0.2mm,             % No line under title
  title filled,              % Fill the title background
  colbacktitle=mygreenlight,     % Background color of the title
  before skip=10pt,          % Space before the box
  after skip=10pt            % Space after the box
}
